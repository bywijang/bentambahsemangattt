  <script>

    AOS.init({ duration: 1200, once: false });

    

    const mainAudio = document.getElementById("mainAudio");

    const musicImg = document.getElementById("music-img");


    function startSurprise() {

      document.getElementById('cover').style.transform = 'translateY(-100%)';

      document.body.style.overflow = 'auto';

      mainAudio.volume = 0.3;

      mainAudio.play();

      musicImg.style.animationPlayState = 'running';

    }


    function toggleMusic() {

      if (mainAudio.paused) {

        mainAudio.play();

        musicImg.style.animationPlayState = 'running';

      } else {

        mainAudio.pause();

        musicImg.style.animationPlayState = 'paused';

      }

    }


    function playSFX(file) {

      const sfx = new Audio(file);

      sfx.volume = 0.5;

      sfx.play();

    }


    // Countdown 4 April 2026 04:04 AM

    const target = new Date("April 4, 2026 04:04:00").getTime();

    setInterval(() => {

      const now = new Date().getTime();

      const diff = target - now;

      if (diff < 0) return;

      document.getElementById("days").innerText = Math.floor(diff / (1000 * 60 * 60 * 24)).toString().padStart(2, '0');

      document.getElementById("hours").innerText = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)).toString().padStart(2, '0');

      document.getElementById("minutes").innerText = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)).toString().padStart(2, '0');

      document.getElementById("seconds").innerText = Math.floor((diff % (1000 * 60)) / 1000).toString().padStart(2, '0');

    }, 1000);


    // Live Message Handle

    // GANTI SELURUH BAGIAN SCRIPT FORM DI DALAM HTML MAS DENGAN INI:

const commentContainer = document.getElementById('liveMessages');

  const inisial = nama.charAt(0).toUpperCase();

      <span class="font-bold text-[10px] text-[#7A5A3D] uppercase tracking-wider">${nama}</span>

    </div>

    <p class="text-[11px] text-stone-500 italic leading-relaxed">"${pesan}"</p>

  `;

  commentContainer.prepend(newComment);
